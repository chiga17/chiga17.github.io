c2dll Library

from c2dll import dllfunc, as_ptr
import numpy as np

def my_func(eps, n):
    d = dllfunc('''
        DLL_API void f(double* params, int n, double* out) {
            for (int i=0; i<n; i++) {
                out[i] = ... // some heavy calculations
            }
        }''')
    n = int(n)
    results = np.zeros(shape=(n))
    d.f(as_ptr([eps,]), n, as_ptr(results))
    return results

print(my_func(0.1, 1000))
